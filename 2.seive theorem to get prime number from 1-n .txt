#include<bits/stdc++.h>
#define M 1000000
bool marked[M];
using namespace std;

void sieve(int n1)
{
    for(int i=3; i*i<=n1; i+=2)
    {
        if(marked[i]==false)
        {
            for(int j=i*i; j<=n1; j+=i+i)
            {
                marked[j]=true;
            }
        }
    }
}

bool isPrime(int n)
{
    if (n < 2)
        return false;
    if (n == 2)
        return true;
    if (n % 2 == 0)
        return false;
    if(marked[n]==false)
        return true;
    else
        return false;

}
int main()
{
    int n,x=0;
    vector<int>v1;
    cin>>n;
    sieve(n);
    for(int i=1; i<=n; i++)
    {
        if(isPrime(i)==true)
            v1.push_back(i);
    }
    x=v1.size();
    for(int i=0; i<x; i++)
    {
        cout<<v1[i]<<" ";

    }
    cout<<endl;
    return 0;
}
