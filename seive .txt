#include<bits/stdc++.h>
#define M 1000000
using namespace std;
bool isPrime[1000001];
void sieve()
{
   isPrime[0]=isPrime[1]=1;
    for(int i=2; i*i<=M; i++)
    {
        if(!isPrime[i])
        {
            for(int j=i*i; j<=M; j+=i)
                isPrime[j]=1;
        }
    }
}
int main()
{
    int i,n;
    cin>>n;
    sieve();
    for(int i=1; i<=n; i++)
    {
        if(isPrime[i])
            cout<<i<<" ";
    }

}
