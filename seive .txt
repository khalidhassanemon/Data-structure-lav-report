#include<bits/stdc++.h>
#define M 1000000
using namespace std;
int isPrime[1000001];
void sieve()
{
    for(int i=1; i<=M; i++)
        isPrime[i]=1;

    isPrime[0]=isPrime[1]=0;
    for(int i=2; i*i<=M; i++)
    {
        if(isPrime[i])
        {
            for(int j=i*i; j<=M; j+=i)
                isPrime[j]=0;
        }
    }
}
int main()
{
    int i,n;
    cin>>n;
    sieve();
    for(int i=1; i<=n; i++)
    {
        if(isPrime[i])
            cout<<i<<" ";
    }

}
